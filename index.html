-- Skills Assessment Query
CREATE INDEX idx_skills_proficiency ON developer_skills(proficiency DESC);

SELECT skill_name, proficiency, years_experience,
  CASE
    WHEN proficiency >= 90 THEN 'Expert'
    WHEN proficiency >= 70 THEN 'Advanced'
    WHEN proficiency >= 50 THEN 'Intermediate'
    ELSE 'Beginner'
  END as level
FROM developer_skills
WHERE developer_id = 'john_doe'
ORDER BY proficiency DESC;
